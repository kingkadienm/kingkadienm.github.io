# 类与对象

class Person:
    def __init__(self, name):
        self.name = name

    def greet(self):
        return f"Hi, {self.name}"

    @classmethod
    def from_birth(cls, name, year):
        return cls(name)

    @staticmethod
    def is_valid_name(name):
        return len(name) > 0

class Student(Person):
    def __init__(self, name, grade):
        super().__init__(name)
        self.grade = grade
