# Python进阶

# 装饰器

def deco(func):
    def wrapper(*args, **kwargs):
        print('before')
        result = func(*args, **kwargs)
        print('after')
        return result
    return wrapper

@deco
def f():
    pass

from functools import wraps
def deco2(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

# 异步 asyncio

import asyncio

async def fetch():
    await asyncio.sleep(1)
    return 1

async def main():
    tasks = [fetch() for _ in range(5)]
    results = await asyncio.gather(*tasks)

asyncio.run(main())
